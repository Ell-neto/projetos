Manipulação de Objetos - Questão 3

Objetivo: Dado um objeto produto com várias propriedades, crie uma função que retorna um novo objeto contendo apenas as propriedades cujo valor seja maior que um valor específico. Use for in para filtrar as propriedades.

                function executarCodigo() {
                    const produto = {
                        nome: 'Notebook',
                        preco: 3000,
                        estoque: 50,
                        garantia: 24, // meses
                        peso: 2.5, // kg
                        desconto: 500,
                        anoFabricacao: 2022
                    };
        
                    // Determinei um valor limite para o filtro
                    const valorLimite = 1000;
        
                    // Função que filtra as propriedades com valores maiores que o valor limite
                    function filtrarPropriedades(obj, limite) {
                        const novoObjeto = {};
                        for (let propriedade in obj) {
                            if (obj[propriedade] > limite) {
                                novoObjeto[propriedade] = obj[propriedade];
                            }
                        }
                        return novoObjeto;
                    }
        
                    // Filtrando as propriedades do objeto produto
                    const produtoFiltrado = filtrarPropriedades(produto, valorLimite);
                }
